    \section*{D: Affine Transformation of Independent Normals}

        \begin{equation}
            x = L z + \mu; \hspace{20pt} z \sim \text{N}(0,1)
        \end{equation}

        \begin{align}
            \E [x] &= \E[Lz] + \E[\mu] = L \E[z] + \mu = \mu
        \end{align}

        \begin{align}
            \cov(x) &= \cov(Lz, Lz) + \cov(Lz, \mu) + \cov(\mu, Lz) + \cov(\mu \mu) \\
            & = L \cov(z, z) L^T = L L^T
        \end{align}
